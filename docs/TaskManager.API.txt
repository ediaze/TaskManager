Environment
-------------------------------------------------------------------------------
# ASPNETCORE_ENVIRONMENT environment variable. 
# e.g.: The value can be Development, Staging, or Production.

# --- cmd ---
set ConnectionStrings__TaskManagerDatabase=YourConnectionStringHere
set DOTNET_SYSTEM_GLOBALIZATION_INVARIANT=0
echo %ConnectionStrings__TaskManagerDatabase%

# --- bash ---
export ConnectionStrings__TaskManagerDatabase=<Your Connection String>
export DOTNET_SYSTEM_GLOBALIZATION_INVARIANT=0

echo $ConnectionStrings__TaskManagerDatabase

# --- Powershell ---
$env:ConnectionStrings__TaskManagerDatabase="YourConnectionStringHere"
echo %ConnectionStrings__TaskManagerDatabase%
